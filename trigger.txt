CREATE OR REPLACE FUNCTION update_blog_views_count()
RETURNS TRIGGER AS $$
BEGIN
    -- Для INSERT - увеличиваем счетчик
    IF (TG_OP = 'INSERT') THEN
        UPDATE blogs 
        SET count_views = COALESCE(count_views, 0) + 1
        WHERE id_blog = NEW.id_blog;
    
    -- Для DELETE - уменьшаем счетчик
    ELSIF (TG_OP = 'DELETE') THEN
        UPDATE blogs 
        SET count_views = GREATEST(COALESCE(count_views, 0) - 1, 0)
        WHERE id_blog = OLD.id_blog;
    END IF;
    
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

-- 2. Создаем триггер на таблицу views
CREATE TRIGGER trigger_update_blog_views
    AFTER INSERT OR DELETE ON views
    FOR EACH ROW
    EXECUTE FUNCTION update_blog_views_count();
